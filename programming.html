FFT Problem 1 Solution

FFT Problem 1 is a Dynamic Programming problem that I have solved. You can see the problem statement here: 

							
import java.util.Scanner;
public class Fft1 {

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int x = input.nextInt();
        long dp[][] = new long [x+1][x+1];
        long arr[] = new long[x+1];

        for (int i = 1; i <= x; i++){
            arr[i] = input.nextLong();
        }
        for (int i = 0; i <= x; i++){
            dp[0][i] = 1;
        }
            for (int i = 1; i <= x; i++){
                for (int j = i; j <= x; j++){
                    dp[i][j] = (((dp[i-1][j-1]*arr[j])% 998244353 + dp[i][j-1])% 998244353 +998244353)%998244353;
                }
            }
            for(int i = 1; i <= x; i++){
                System.out.print(dp[i][x] + " ");
            }
        
        
    }
    
}
